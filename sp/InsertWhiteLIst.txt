ALTER PROCEDURE [dbo].[InsertWhiteListedProjects]
AS
BEGIN
    SET NOCOUNT ON;
    WITH RankedProjects
    AS   (SELECT b.PsmId AS Id,
                 u.Appointment AS Appt,
                 br.unitname AS Fmn,
                 u.Tele_Army AS ContactNo,
                 h.HostTypeID AS mHostTypeId,
                 h.HostingDesc AS HostedOn,
                 b.CertNo,
                 b.DateTimeOfUpdate AS DDGIT_WL_DT,
                 p.ProjName,
                 p.InitialRemark AS Remarks,
                 acg.DateTimeOfUpdate,
                 b.TimeStamp,
                 b.StatusActionsMappingId,
                 ROW_NUMBER() OVER (PARTITION BY p.ProjName ORDER BY b.PsmId DESC) AS rn
          FROM   Projects AS p
                 INNER JOIN
                 ProjStakeHolderMov AS b
                 ON p.ProjId = b.ProjId
                 INNER JOIN
                 tbl_mUnitBranch AS br
                 ON br.unitid = p.StakeHolderId
                 INNER JOIN
                 AspNetUsers AS u
                 ON p.StakeHolderId = u.unitid
                 LEFT OUTER JOIN
                 mHostType AS h
                 ON p.HostTypeID = h.HostTypeID OUTER APPLY (SELECT   TOP 1 DateTimeOfUpdate
                                                             FROM     ProjStakeHolderMov
                                                             WHERE    ProjId = p.ProjId
                                                                      AND StatusActionsMappingId = 78
                                                             ORDER BY DateTimeOfUpdate DESC) AS acg
          WHERE  b.StatusActionsMappingId IN (78, 88)
                 AND p.ProjName NOT IN (SELECT ProjName
                                        FROM   trnWhiteListed))
    INSERT INTO trnWhiteListed (mHostTypeId, Appt, Fmn, ContactNo, Clearence, CertNo, ValidUpto, Remarks, ProjName, Date, IsWhiteListed)
    SELECT rp.mHostTypeId,
           rp.Appt,
           rp.Fmn,
           rp.ContactNo,
           rp.DateTimeOfUpdate AS Clearence,
		     
           CASE WHEN rp.StatusActionsMappingId = 88 THEN rp.DDGIT_WL_DT ELSE '' END AS CertNo,
           CASE WHEN rp.StatusActionsMappingId = 88 THEN rp.DDGIT_WL_DT ELSE '' END AS ValidUpto,
           rp.Remarks,
           rp.ProjName,
           rp.TimeStamp,
           CASE WHEN rp.StatusActionsMappingId = 88 THEN 1 ELSE 0 END AS IsWhiteListed
    FROM   RankedProjects AS rp
    WHERE  rp.rn = 1
           AND NOT EXISTS (SELECT 1
                           FROM   trnWhiteListed AS tw
                           WHERE  tw.ProjName = rp.ProjName);
END
