ALTER PROCEDURE [dbo].[GetWhiteListedProjects]
AS
BEGIN
    SET NOCOUNT ON;

    WITH RankedProjects AS (
        SELECT
            b.PsmId AS Id,
            u.appointment,
            p.ProjId,
            b.CertNo,
            p.ProjName,
            p.Sponsor AS Sponser,
            u.Tele_Army AS ContactNo,
            h.HostingDesc AS HostedOn,
            acg.DateTimeOfUpdate,
            p.InitialRemark AS Remarks,
            ROW_NUMBER() OVER (PARTITION BY p.ProjName ORDER BY b.PsmId DESC) AS rn
        FROM Projects p
        INNER JOIN ProjStakeHolderMov b ON p.ProjId = b.ProjId
        INNER JOIN AspNetUsers u ON p.StakeHolderId = u.unitid
        LEFT JOIN mHostType h ON p.HostTypeID = h.HostTypeID
        OUTER APPLY (
            SELECT TOP 1 DateTimeOfUpdate
            FROM ProjStakeHolderMov
            WHERE ProjId = p.ProjId AND StatusActionsMappingId = 78
            ORDER BY DateTimeOfUpdate DESC
        ) AS acg
        WHERE b.StatusActionsMappingId = 88
    )
    SELECT
        Id,
        ProjId,
        Appointment,
        CertNo,
        ProjName,
        HostedOn,
        Sponser,
        ContactNo,
        FORMAT(DateTimeOfUpdate, 'dd/MM/yyyy') AS ACGClearence,
        FORMAT(DATEADD(YEAR, 3, DateTimeOfUpdate), 'dd/MM/yyyy') AS ValidUpto,
        Remarks
    FROM RankedProjects
    WHERE rn = 1
	
END
