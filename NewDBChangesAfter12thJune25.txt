ALTER PROCEDURE [dbo].[usp_GetActInbox]
    @UnitId INT
AS
BEGIN
    SET NOCOUNT ON;

    SELECT
        a.ProjId,
        b.PsmId AS PsmIds,
        a.ProjName,
        a.StakeHolderId,
        stackcs.UnitName AS StakeHolder,
        s.Status,
        s.StatusId,
        b.FromUnitId,
        b.ToUnitId,
        toUnit.UnitName AS ToUnitName,
        act.Actions AS Action,
        0 AS TotalDays,
        map.ActionsId AS ActionId,
        a.Sponsor,
        ISNULL(fromUnit.UnitName, '') AS UnitName,
        fromUnit.UnitName AS FromUnitUserDetail,
        fromUnit.UnitName + ' ( ' + ISNULL(b.UserDetails, '') + ')' AS FromUnitName,
        stg.Stages AS Stage,
        stg.StagesId AS StageId,
        b.IsRead,
        a.IsProcess,
        b.TimeStamp,
        
        -- Date_type field
        a.Date_type,

        -- AdminApprovalStatus field (Check for DDGIT_approval == true)
        CASE 
            WHEN adminApproval.DDGIT_approval = 1 THEN 1
            ELSE 0
        END AS AdminApprovalStatus,

        -- UserRequest field (Check if any DateApproval record exists with DDGIT_approval false or null)
        CASE 
            WHEN EXISTS ( 
                SELECT 1 
                FROM DateApproval da 
                WHERE da.ProjId = a.ProjId AND (da.DDGIT_approval = 0 OR da.DDGIT_approval IS NULL)
            ) THEN 1
            ELSE 0
        END AS UserRequest,

		   -- LatestActionType field from LegacyHistory
        (
            SELECT TOP 1 lh.ActionType
            FROM LegacyHistory lh
            WHERE lh.ProjectId = a.ProjId
            ORDER BY lh.ActionDate DESC
        ) AS LatestActionType,

        -- RequestUnitId field (Fetch the UnitId where DDGIT_approval is false or null, and order by Request_Date)
        (SELECT TOP 1 da.UnitId
         FROM DateApproval da
         WHERE da.ProjId = a.ProjId AND (da.DDGIT_approval = 0 OR da.DDGIT_approval IS NULL)
         ORDER BY da.Request_Date DESC) AS RequestUnitId

    FROM Projects a
    JOIN ProjStakeHolderMov b ON a.ProjId = b.ProjId
    LEFT JOIN tbl_mUnitBranch stackcs ON a.StakeHolderId = stackcs.unitid
    LEFT JOIN DateApproval adminApproval ON a.ProjId = adminApproval.ProjId AND adminApproval.DDGIT_approval = 1
    JOIN TrnStatusActionsMapping map ON b.StatusActionsMappingId = map.StatusActionsMappingId
    JOIN mStatus s ON map.StatusId = s.StatusId
    JOIN mActions act ON map.ActionsId = act.ActionsId
    LEFT JOIN tbl_mUnitBranch toUnit ON b.ToUnitId = toUnit.unitid
    LEFT JOIN tbl_mUnitBranch fromUnit ON b.FromUnitId = fromUnit.unitid
    LEFT JOIN mStages stg ON s.StageId = stg.StagesId
    WHERE a.IsActive = 1 AND a.IsDeleted = 0
      AND b.IsActive = 1 AND b.IsDeleted = 0
      AND a.IsSubmited = 1 AND b.IsComplete = 0
      AND b.ToUnitId = @UnitId AND b.IsComment = 0
    ORDER BY b.TimeStamp DESC;
END
